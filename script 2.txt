function serverAuth() {
    fetch(`${BACKEND_URL}/api/profile/${myTelegramId}`)
    .then(res => res.json())
    .then(data => {
        score = data.score;
        currentEnergy = data.energy;
        maxEnergy = data.max_energy || 400;
        clickPower = data.click_power || 0.0025;
        currentUpgradeLevel = data.upgrade_level || 0;

        scoreEl.innerText = score.toFixed(4);
        updateEnergyUI();
        updateShopUI();
        console.log("Успешно синхронизировано с Python БД!");
    })
    .catch(err => console.error("Ошибка подключения к серверу БД:", err));
}
